.data

top: .word 0
str: .asciiz "sayi : "



.text
.globl main

main: 

lui $t0, 4097		#t0 başlangıç adresi yapıldı
addi $t1, $zero, 10	#10 sayı okunacağı belirlendi	
addi $t2, $zero, 1	#ilk sayı okunacak
lw $s1, 0($t0)		# top değeri başlangıç değeri s1'de
loop:li $v0, 4
la $a0, str
syscall
li $v0, 5
syscall
add $s1, $s1, $v0
addi $t2, $t2, 1
ble $t2, $t1, loop
sw $s1, 0($t0)

add $a0, $s1, $zero
li $v0, 1
syscall


li $v0, 10
syscall
.end main
